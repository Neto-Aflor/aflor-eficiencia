    <script>
        let myChart;


        function calcular() {
            // Entradas
            const n = parseFloat(document.getElementById("colab").value);
            const salario = parseFloat(document.getElementById("salario").value);
            const dias = parseFloat(document.getElementById("dias").value);
            const custoHora = salario / 220;
            const custoMinuto = custoHora / 60;


            // Perdas Manuais (Base Planilha)
            const custoPapel = (n * parseFloat(document.getElementById("numForm").value) * parseFloat(document.getElementById("custoForm").value)) * dias;
            const tempoTransporte = n * parseFloat(document.getElementById("minTransp").value) * custoMinuto * dias;
            const tempoDigitalizacao = n * parseFloat(document.getElementById("minDigit").value) * custoMinuto * dias;
            const tempoRastreio = n * parseFloat(document.getElementById("minRastreio").value) * custoMinuto * dias;
            const tempoRelatorio = n * parseFloat(document.getElementById("minRelat").value) * custoMinuto * dias;


            const custoManualTotal = custoPapel + tempoTransporte + tempoDigitalizacao + tempoRastreio + tempoRelatorio;


            // Preços AFLOR (Base Tabela ROI fornecida)
            let aflor = 0;
            if (n === 1) aflor = 550;
            else if (n === 2) aflor = 650;
            else if (n === 3) aflor = 750;
            else if (n === 4) aflor = 850;
            else aflor = n * 185; // Acima de 4 conforme solicitado


            const economiaMensal = custoManualTotal - aflor;
            const roi = (economiaMensal / aflor) * 100;
            const ganhoAnual = economiaMensal * 12;
            const custoEvitadoPerc = (economiaMensal / custoManualTotal) * 100;
            const payback = aflor / economiaMensal;


            // UI
            document.getElementById("ganhoAnual").innerText = ganhoAnual.toLocaleString('pt-BR', {style: 'currency', currency: 'BRL'});
            document.getElementById("roi").innerText = roi.toFixed(1) + "%";
            document.getElementById("evitado").innerText = custoEvitadoPerc.toFixed(1) + "%";
            document.getElementById("payback").innerText = payback.toFixed(2) + " meses";


            // Storytelling
            document.getElementById("story").innerHTML = `
                <p>Hoje, sua ineficiência operacional custa <b class="text-red-400">R$ ${custoManualTotal.toLocaleString('pt-BR', {minimumFractionDigits:2})} por mês</b>.</p>
                <p>Com a <b>AFLOR</b>, você substitui papel e retrabalho por um investimento de <b class="text-green-400">R$ ${aflor.toLocaleString('pt-BR', {minimumFractionDigits:2})}</b>.</p>
                <p>A cada R$ 1 investido, você retorna <b class="text-white">R$ ${(economiaMensal/aflor).toFixed(1)}</b> direto para o seu caixa, totalizando <b class="text-green-500">R$ ${ganhoAnual.toLocaleString('pt-BR', {maximumFractionDigits:0})}</b> de lucro líquido no primeiro ano.</p>
            `;


            updateChart(custoManualTotal, aflor);
        }


        function updateChart(manual, aflor) {
            const ctx = document.getElementById('roiChart').getContext('2d');
            if (myChart) myChart.destroy();
            myChart = new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: ['PROCESSO MANUAL', 'AFLOR'],
                    datasets: [{
                        data: [manual, aflor],
                        backgroundColor: ['#ef4444', '#22c55e'],
                        borderRadius: 6,
                        barThickness: 60
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: { legend: { display: false } },
                    scales: { 
                        y: { beginAtZero: true, grid: { color: 'rgba(255,255,255,0.05)' }, ticks: { color: '#64748b' } },
                        x: { grid: { display: false }, ticks: { color: 'white', font: { weight: 'bold' } } }
                    }
                }
            });
        }


        function downloadHTML() {
            const html = document.documentElement.outerHTML;
            const blob = new Blob([html], {type: 'text/html'});
            const a = document.createElement('a');
            a.href = URL.createObjectURL(blob);
            a.download = 'AFLOR_ROI_Dashboard.html';
            a.click();
        }


        window.onload = calcular;
    </script>
